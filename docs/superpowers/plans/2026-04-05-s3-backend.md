# S3 Compatible Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an S3-compatible storage backend to mizchi/vfs with async trait support, using `moonbitlang/async` for HTTP and Promise integration.

**Architecture:** Async-ify all VFS traits (breaking change). Add `moonbitlang/async` dependency for HTTP client (`@http.Client`) and async runtime. Build S3 client layer on top of `@http.Client` with SigV4 signing via JS FFI. Wrap S3 client as a VFS `FileSystem`/`ReadableFileSystem` adapter.

**Tech Stack:** MoonBit, moonbitlang/async (HTTP client, io, js_async), JS `crypto.subtle` for HMAC-SHA256 signing.

**Key Discovery:** `moonbitlang/async` already provides `@http.Client` with `get()`, `put()`, `post()`, `request()`, `end_request()` — plus `@io.Reader::read_all()` for response bodies. No custom HTTP FFI needed. Only SigV4 signing requires FFI.

/// Compute SHA-256 hex digest of bytes.
pub async fn sha256_hex(data : Bytes) -> String {
  sha256_hex_ffi(data).wait()
}

///|
/// Compute HMAC-SHA256.
pub async fn hmac_sha256(key : Bytes, data : Bytes) -> Bytes {
  hmac_sha256_ffi(key, data).wait()
}

///|
/// Format a date as YYYYMMDD.
fn date_stamp(date : String) -> String {
  // date is ISO 8601: "20260405T120000Z"
  // return first 8 chars
  date[:8].to_string()
}

///|
/// Build the AWS SigV4 signing key.
async fn derive_signing_key(
  secret : String,
  date_stamp : String,
  region : String,
  service : String,
) -> Bytes {
  let k_secret = @encoding.utf8.encode("AWS4" + secret)
  let k_date = hmac_sha256(k_secret, @encoding.utf8.encode(date_stamp))
  let k_region = hmac_sha256(k_date, @encoding.utf8.encode(region))
  let k_service = hmac_sha256(k_region, @encoding.utf8.encode(service))
  hmac_sha256(k_service, @encoding.utf8.encode("aws4_request"))
}

///|
/// Sign an S3 request using AWS Signature Version 4.
/// Returns the Authorization header value and additional headers to add.
pub async fn sign_request(
  config : S3Config,
  method : String,
  path : String,
  query : String,
  headers : Array[(String, String)],
  payload_hash : String,
  amz_date : String,
) -> (String, Array[(String, String)]) {
  let ds = date_stamp(amz_date)
  let credential_scope = "\{ds}/\{config.region}/s3/aws4_request"
  // Build canonical headers (sorted by lowercase key)
  let sorted_headers = headers.copy()
  sorted_headers.sort_by(fn(a, b) { a.0.compare(b.0) })
  let canonical_headers = StringBuilder::new()
  let signed_header_names : Array[String] = []
  for pair in sorted_headers {
    let (k, v) = pair
    canonical_headers.write_string("\{k}:\{v}\n")
    signed_header_names.push(k)
  }
  let signed_headers = signed_header_names.join(";")
  // Canonical request
  let canonical_request =
    [method, path, query, canonical_headers.to_string(), signed_headers, payload_hash].join(
      "\n",
    )
  let canonical_request_hash = sha256_hex(@encoding.utf8.encode(canonical_request))
  // String to sign
  let string_to_sign =
    ["AWS4-HMAC-SHA256", amz_date, credential_scope, canonical_request_hash].join(
      "\n",
    )
  // Signing key and signature
  let signing_key = derive_signing_key(
    config.secret_access_key,
    ds,
    config.region,
    "s3",
  )
  let signature_bytes = hmac_sha256(
    signing_key,
    @encoding.utf8.encode(string_to_sign),
  )
  let signature = bytes_to_hex(signature_bytes)
  let authorization = "AWS4-HMAC-SHA256 Credential=\{config.access_key_id}/\{credential_scope}, SignedHeaders=\{signed_headers}, Signature=\{signature}"
  let extra_headers : Array[(String, String)] = [
    ("authorization", authorization),
  ]
  (authorization, extra_headers)
}

///|
/// Convert bytes to lowercase hex string.
fn bytes_to_hex(b : Bytes) -> String {
  let sb = StringBuilder::new()
  let hex = "0123456789abcdef"
  for i in 0..<b.length() {
    let byte = b[i].to_int()
    sb.write_char(hex[byte.lsr(4).land(0xf)].unwrap())
    sb.write_char(hex[byte.land(0xf)].unwrap())
  }
  sb.to_string()
}
```

///|
/// Minimal XML tag value extractor.
/// Finds all occurrences of <tag>value</tag> and returns values.
pub fn extract_tag_values(xml : String, tag : String) -> Array[String] {
  let results : Array[String] = []
  let open_tag = "<\{tag}>"
  let close_tag = "</\{tag}>"
  let mut pos = 0
  while pos < xml.length() {
    let start = string_index_of(xml, open_tag, pos)
    if start < 0 {
      break
    }
    let content_start = start + open_tag.length()
    let end = string_index_of(xml, close_tag, content_start)
    if end < 0 {
      break
    }
    results.push(xml.substring(start=content_start, end~))
    pos = end + close_tag.length()
  }
  results
}

///|
/// Find substring index starting from pos. Returns -1 if not found.
fn string_index_of(haystack : String, needle : String, from : Int) -> Int {
  let h_len = haystack.length()
  let n_len = needle.length()
  if n_len == 0 {
    return from
  }
  if from + n_len > h_len {
    return -1
  }
  for i in from..=(h_len - n_len) {
    let mut matched = true
    for j in 0..<n_len {
      if haystack[i + j] != needle[j] {
        matched = false
        break
      }
    }
    if matched {
      return i
    }
  }
  -1
}

///|
/// Parse ListObjectsV2 XML response into S3Object array.
pub fn parse_list_objects_v2(xml : String) -> Array[S3Object] {
  let keys = extract_tag_values(xml, "Key")
  let sizes = extract_tag_values(xml, "Size")
  let results : Array[S3Object] = []
  for i in 0..<keys.length() {
    let size : Int64 = if i < sizes.length() {
      try {
        @strconv.parse_int64(sizes[i])
      } catch {
        _ => 0L
      }
    } else {
      0L
    }
    results.push({ key: keys[i], size })
  }
  results
}
```

async fn S3Client::ensure_client(self : S3Client) -> @http.Client {
  match self.client {
    Some(c) => c
    None => {
      let c = @http.Client::new(self.config.endpoint)
      self.client = Some(c)
      c
    }
  }
}

///|
/// Build signed headers for an S3 request.
async fn S3Client::signed_headers(
  self : S3Client,
  method : String,
  key : String,
  query : String,
  payload_hash : String,
) -> Map[String, String] {
  let amz_date = amz_date_now()
  let host = extract_host(self.config.endpoint)
  let headers : Array[(String, String)] = [
    ("host", host),
    ("x-amz-content-sha256", payload_hash),
    ("x-amz-date", amz_date),
  ]
  let (auth, _) = sign_request(
    self.config,
    method,
    self.signing_path(key),
    query,
    headers,
    payload_hash,
    amz_date,
  )
  let result : Map[String, String] = {}
  result["x-amz-content-sha256"] = payload_hash
  result["x-amz-date"] = amz_date
  result["authorization"] = auth
  result
}

///|
/// Extract host from URL (e.g. "https://example.com:8080" -> "example.com:8080").
fn extract_host(url : String) -> String {
  let s = if url is [.. "https://", .. rest] {
    rest.to_string()
  } else if url is [.. "http://", .. rest] {
    rest.to_string()
  } else {
    url
  }
  // Strip trailing path
  let idx = string_index_of(s, "/", 0)
  if idx >= 0 {
    s.substring(end=idx)
  } else {
    s
  }
}

/// Get an object from S3.
pub async fn S3Client::get_object(
  self : S3Client,
  key : String,
) -> Bytes raise @vfs.VfsError {
  let payload_hash = "e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855" // SHA256 of empty
  let extra = self.signed_headers("GET", key, "", payload_hash)
  let client = self.ensure_client()
  let response = client.get("/\{self.config.bucket}/\{key}", extra_headers=extra)
  if response.code != 200 {
    raise @vfs.vfs_error("S3 GET failed: status \{response.code}")
  }
  let data = (client : &@io.Reader).read_all()
  data.to_bytes()
}

///|
/// Put an object to S3.
pub async fn S3Client::put_object(
  self : S3Client,
  key : String,
  data : Bytes,
) -> Unit raise @vfs.VfsError {
  let payload_hash = sha256_hex(data)
  let extra = self.signed_headers("PUT", key, "", payload_hash)
  let client = self.ensure_client()
  let response = client.put(
    "/\{self.config.bucket}/\{key}",
    data,
    extra_headers=extra,
  )
  if response.code != 200 && response.code != 204 {
    raise @vfs.vfs_error("S3 PUT failed: status \{response.code}")
  }
}

///|
/// Head an object (check existence and get metadata).
pub async fn S3Client::head_object(
  self : S3Client,
  key : String,
) -> S3Object? raise @vfs.VfsError {
  let payload_hash = "e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855"
  let extra = self.signed_headers("HEAD", key, "", payload_hash)
  let client = self.ensure_client()
  client.request(Head, "/\{self.config.bucket}/\{key}", extra_headers=extra)
  let response = client.end_request()
  if response.code == 404 {
    return None
  }
  if response.code != 200 {
    raise @vfs.vfs_error("S3 HEAD failed: status \{response.code}")
  }
  let size : Int64 = match response.headers.get("content-length") {
    Some(s) =>
      try {
        @strconv.parse_int64(s)
      } catch {
        _ => 0L
      }
    None => 0L
  }
  Some({ key, size })
}

pub struct S3Fs {
  client : @s3client.S3Client
  prefix : String
}

///|
pub fn S3Fs::new(client : @s3client.S3Client, prefix? : String = "") -> S3Fs {
  { client, prefix }
}

///|
/// Convert VFS path to S3 key.
/// "/foo/bar.txt" -> "prefix/foo/bar.txt" (strip leading /)
fn S3Fs::to_key(self : S3Fs, path : String) -> String {
  let normalized = @path.normalize(path)
  let stripped = if normalized is ['/', .. rest] {
    rest.to_string()
  } else {
    normalized
  }
  if self.prefix.is_empty() {
    stripped
  } else {
    "\{self.prefix}\{stripped}"
  }
}

///|
/// Convert VFS directory path to S3 directory key (trailing /).
fn S3Fs::to_dir_key(self : S3Fs, path : String) -> String {
  let normalized = @path.normalize(path)
  if normalized == "/" {
    return self.prefix
  }
  let key = self.to_key(normalized)
  if key.is_empty() {
    ""
  } else {
    "\{key}/"
  }
}

///|
/// Ensure all ancestor directories exist as 0-byte marker objects.
async fn S3Fs::ensure_dir_chain(self : S3Fs, path : String) -> Unit raise @vfs.VfsError {
  let normalized = @path.normalize(path)
  if normalized == "/" {
    return
  }
  let parts = split_path(normalized)
  let mut current = ""
  for part in parts {
    current = if current.is_empty() { part } else { "\{current}/\{part}" }
    let dir_key = if self.prefix.is_empty() {
      "\{current}/"
    } else {
      "\{self.prefix}\{current}/"
    }
    // Only create if not exists (avoid unnecessary PUTs)
    let existing = self.client.head_object(dir_key)
    if existing is None {
      self.client.put_object(dir_key, b"")
    }
  }
}

///|
/// Split a normalized path into parts (without leading empty string).
fn split_path(path : String) -> Array[String] {
  let parts : Array[String] = []
  let buf = StringBuilder::new()
  for ch in path {
    if ch == '/' {
      let s = buf.to_string()
      if not(s.is_empty()) {
        parts.push(s)
      }
      buf.reset()
    } else {
      buf.write_char(ch)
    }
  }
  let last = buf.to_string()
  if not(last.is_empty()) {
    parts.push(last)
  }
  parts
}

///|
impl @vfs.ReadableFileSystem for S3Fs with readdir(self, path) {
  let dir_key = self.to_dir_key(path)
  let objects = self.client.list_objects(dir_key)
  let entries : Array[String] = []
  let dir_key_len = dir_key.length()
  for obj in objects {
    let relative = obj.key.substring(start=dir_key_len)
    if relative.is_empty() {
      continue
    }
    // Only immediate children: no "/" in the relative part (except trailing)
    let slash_idx = string_index_of(relative, "/", 0)
    if slash_idx < 0 {
      // File
      entries.push(relative)
    } else if slash_idx == relative.length() - 1 {
      // Directory (trailing /)
      entries.push(relative.substring(end=slash_idx))
    }
    // else: nested object, skip
  }
  entries
}

///|
fn string_index_of(haystack : String, needle : String, from : Int) -> Int {
  let h_len = haystack.length()
  let n_len = needle.length()
  if n_len == 0 {
    return from
  }
  if from + n_len > h_len {
    return -1
  }
  for i in from..=(h_len - n_len) {
    let mut matched = true
    for j in 0..<n_len {
      if haystack[i + j] != needle[j] {
        matched = false
        break
      }
    }
    if matched {
      return i
    }
  }
  -1
}

  if normalized == "/" {
    return true
  }
  let dir_key = self.to_dir_key(path)
  match try? self.client.head_object(dir_key) {
    Ok(Some(_)) => true
    _ => false
  }
}

    Ok(Some(_)) => true
    _ => false
  }
}

///|
impl @vfs.Snapshottable for S3Fs with export_snapshot(self) {
  let all_objects = self.client.list_objects(self.prefix)
  let dirs : Array[String] = ["/"]
  let files : Array[@vfs.SnapshotFile] = []
  let prefix_len = self.prefix.length()
  for obj in all_objects {
    let relative = obj.key.substring(start=prefix_len)
    if relative.is_empty() {
      continue
    }
    let vfs_path = "/\{relative}"
    if vfs_path is [.., '/'] {
      // Directory marker
      let dir_path = vfs_path.substring(end=vfs_path.length() - 1)
      if not(dir_path.is_empty()) && dir_path != "/" {
        dirs.push(dir_path)
      }
    } else {
      // File
      let data = self.client.get_object(obj.key)
      files.push(@vfs.snapshot_file(vfs_path, data))
    }
  }
  @vfs.snapshot(dirs, files)
}

///|
impl @vfs.Snapshottable for S3Fs with replace_snapshot(self, snapshot) {
  // Delete all existing objects
  let existing = self.client.list_objects(self.prefix)
  for obj in existing {
    self.client.delete_object(obj.key)
  }
  // Recreate from snapshot
  for dir in snapshot.dirs {
    if dir != "/" {
      self.ensure_dir_chain(dir)
    }
  }
  for file in snapshot.files {
    let parent = @path.parent(@path.normalize(file.path))
    self.ensure_dir_chain(parent)
    let key = self.to_key(file.path)
    self.client.put_object(key, file.data)
  }
}
